def self.last_incomplete_order user_id
  user = User.includes(:orders).find(user_id)
  last_order = user.orders.last
  if last_order.present?
    if last_order.status.completed?
      user.orders.create(delivery: 0, total:0)
    else
      return last_order
    end
  else
    user.orders.create(delivery: 0, total:0)
  end
end
